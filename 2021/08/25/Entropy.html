Calculate the entropy of our arrrays 

def entropy(arr):
  ent = 0
  probs = dict()
  ctr = Counter(arr)
  for e in ctr:
    probs[f'P_{e}'] = ctr[e] / len(arr)
  for p in probs:
    ent += -1 * probs[p] * np.log2(probs[p])
  return round(ent, 2)
